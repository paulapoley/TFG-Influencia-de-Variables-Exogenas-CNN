# Definir la entrada para la imagen
image_input = Input(shape=(64, 64, 3), name='input_image')

output = Dense(7, activation='softmax', name='output')(x)

# Crear el modelo combinado fuera del bucle
model = Model(inputs=image_input, outputs=output)
